### #10809 알파벳 찾기
```python
c = input()
a = 'abcdefghijklmnopqrstuvwxyz'
r = ['-1']*len(a)
for i in range(len(a)):
    for j in range(len(c)):
        if c[j] == a[i]:
            if r[i] == '-1':
                r[i] = '{}'.format(j)
r = ' '.join(r)
print(r)
```

### #2675 문자열 반복
```python
import sys
n = int(input())
for idx in range(n):
    data = sys.stdin.readline().split()
    R = int(data[0])
    S = data[1]
    result = ''
    for i in S:
        for j in range(R):
            result = result + i
    print(result)
```

### #1157 단어 공부
```python
input_data = input()
c = input_data.lower()
a = 'abcdefghijklmnopqrstuvwxyz'
n = [0]*26
## 알파벳을 다 돌면서 나온 횟수 저장
for i in range(26):
    for j in c:
        if j == a[i]:
            n[i] += 1

m = max(n)
cnt = 0
result = ''
# 알파벳 카운드 리스트를 돌면서 맥스값에 해당하는 알파벳 반환
# 맥스가 2개 이상이면 ? 출력
for i in range(26):
    if n[i] == m:
        cnt += 1
        result = a[i].upper()
if cnt > 1:
    print('?')
else:
    print(result)
```
```python
input_data = input()
c = input_data.upper()

# 중복값을 줄여 반복문을 줄인다.
alpa_list = list(set(c))
length = len(alpa_list)
n = [0]*length
# 카운트를 써서 나온 횟수 저장
for i in range(length):
    n[i] = c.count(alpa_list[i])

m = max(n)
if n.count(m) > 1:
    print('?')
else:
    # 알파벳과 n의 저장 위치가 같으므로 인덱스를 사용 가능
    print(alpa_list[n.index(m)])
```
