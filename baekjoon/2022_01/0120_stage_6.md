### #15596 정수 N개의 합
```python
def solve(a):
    ans = 0
    for i in a:
        ans += i
    return ans
```

### #4673 셀프 넘버
```python
def d(n):
    n = n + sum(map(int, str(n)))
    return n

t = set( i for i in range(1, 10000 +1))
s = set()

for idx in range(1, 10000 +1):
    n = idx
    n = d(n)
    if n > 10000:
        continue
    s = s | {n}

result = sorted(t - s)
for i in result:
    print(i)
```

### #1065 한수
```python
def check(n):
    h = n // 100
    t = n // 10 - h*10
    u = n % 10

    # 등차수열이면 1 반환
    if (h-t) == (t-u):
        return 1
    else:
        return 0

def hundreds(n):
    cnt = 0
    for i in range(100, n+1):
        cnt += check(i)
    return cnt + 99

N = int(input())

# 99까지는 모두 한수 이다.
if N < 100:
    print(N)
else:
    print(hundreds(N))
```
