### #5622 다이얼
```python
c = input()
alpha = ['ABC', 'DEF', 'GHI', 'JKL', 'MNO', 'PQRS', 'TUV', 'WXYZ']
d = {}
for i in range(len(alpha)):
    d.update({alpha[i]: i+3})

total = 0

for i in c:
    for j in d.keys():
        if i in j:
            total += d[j]

print(total)
```

### #2941 크로아티아 알파벳
```python
c = input()
cro = ['c=', 'c-', 'dz=', 'd-', 'lj', 'nj', 's=', 'z=']

total = 0
for i in cro:
    total += c.count(i)
    c = c.replace(i, ' ', c.count(i))
c = c.replace(' ', '')
total += len(c)
print(total)
```



### #1316 그룹 단어 체커
```python
n = int(input())
total = 0
for i in range(n):
    c = input()
    a = c[0]
    b = ''
    # 입력받은 문자를 돌다가
    for j in c:
        # 다른 문자가 나오면
        if a != j:
            # 문자열 추가
            b = b+ a
            # a를 새로운 문자열로 바꾸고
            a = j
            # 중복 문자열인지 검사
            if a in b:
                break
    else:
        total += 1
print(total)

```
