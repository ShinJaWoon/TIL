## 0215 IM 대비 문제풀기
### #2578 빙고
https://www.acmicpc.net/problem/2578
```python
# 빙고판을 2차원 리스트에 저장
board = [list(map(int, input().split())) for r in range(5)]
# 사회자가 부르는 번호를 1차원 리스트에 저장
MC = []
for _ in range(5):
    MC.extend(list(map(int, input().split())))

# 빙고 개수 체크 함수. 0이 사회자가 부른 숫자
def check_bingo(board):
    bingo = 0
    # 가로줄 검색
    bingo += board.count([0]*5)
    # 세로줄 검색. 전치 행렬을 통해 [0, 0, 0, 0, 0]을 찾는다.
    board_T = list(map(list, (zip(*board))))
    bingo += board_T.count([0]*5)
    # 대각선 검색
    # 왼쪽 위에서 오른쪽 아래로
    for i in range(5):
        if board[i][i] != 0:
            break
    else:
        bingo += 1
    # 오른쪽 위에서 왼쪽 아래로
    for i in range(5):
        if board[i][4-i] != 0:
            break
    else:
        bingo += 1
    # 빙고 횟수가 3 이상이면 True 아니면 False
    if bingo >= 3:
        return True
    else:
        return False

cnt = 0
# 사회자가 번호를 부르면 맞는 번호를 찾아 0으로 바꾼다.
for num in MC:
    cnt += 1
    for r in range(5):
        for c in range(5):
            if board[r][c] == num:
                board[r][c] = 0
    # 빙고가 3개 이상되면 카운트를 출력하고 break
    if check_bingo(board):
        print(cnt)
        break
```
