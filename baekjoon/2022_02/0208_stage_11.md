### #1018 체스판 다시 칠하기
https://www.acmicpc.net/problem/1018
```python
import sys

N, M = map(int, sys.stdin.readline().split())
board = list()

for row in range(N):
    board.append(sys.stdin.readline().strip())

# 칠하는 횟수를 저장하는 리스트
brush = []

for row in range(0, N-7):
    for col in range(0, M-7):
        board_str = ''
        for row8 in range(8):
            # 한줄은 그대로, 한줄은 리버스로 받음 (지그재그를 검사하기 위함)
            if row8 % 2:
                board_str += board[row+row8][col:col+8][::-1]
            else:
                board_str += board[row+row8][col:col+8]

        # 첫번째가 검정, 두번째가 흰색
        cnt= 0
        for b in range(0, 64, 2):
            if board_str[b] != 'B':
                cnt += 1
            if board_str[b+1] != 'W':
                cnt += 1
        brush.append(cnt)
        # 첫번째가 흰색인 경우는 첫번째가 검정인 경우에서 틀린 경우만 색칠하면 됨
        brush.append(64-cnt)

print(min(brush))
```

### #1436 영화감독 숌
https://www.acmicpc.net/problem/1436
```python
N = int(input())
n = 1
N_1 = N+1
d = dict()
for i in range(0, N):
    if n == N+1:
        break
    i_str = str(i)
    i_len = len(i_str)

    i_str_re = i_str[::-1]
    if i_str[-3:] == '666':
        for k in range(0, 10**(3)):
            d.update({n: f'{i_str}{k:0>3d}'})
            n += 1
        continue
    elif i_str_re[:2] == '66':
        for k in range(0, 10**(2)):
            d.update({n: f'{i_str}6{k:0>2d}'})
            n += 1
        continue
    elif i_str_re[:1] == '6':
        for k in range(0, 10):
            d.update({n: i_str+'66'+str(k)})
            n += 1
        continue
    else:
        d.update({n: i_str+'666'})
        n += 1
title = int(d[N])
print(title)
```
